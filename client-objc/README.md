## Getting Started

```objc


NSNumber* *version = 3.16; // 
NSString* *appKey = @"appKey_example"; // The application key to use when creating an analytic or service request. The account needs to have permissions to the applicaton or it will be denied.
NSString* *name = @"name_example"; // The name of the queue to connect to
NSString* *hostname = @"hostname_example"; // The hostname of the server the queue is hosted on
NSString* *username = @"username_example"; // The username to access the server the queue is hosted on
NSString* *password = @"password_example"; // The password to access the queue to connect to
NSString* *dataMapping = @"dataMapping_example"; // The data mapping information in the format of AMQPRequest
NSString* *deviceId = @"deviceId_example"; // The client deviceID (optional)
NSNumber* *accountId = @56; // The logged in user ID (optional)
NSNumber* *port = @5672; // The port of the server the queue is hosted on (optional) (default to @5672)
NSString* *virtualHost = @"virtualHost_example"; // The virtual host defined on the server the queue is associated on (optional)
NSString* *exchanger = @"exchanger_example"; // The exchanger of the queue to connect to (optional)
NSString* *exchangerType = @"exchangerType_example"; // The exchanger type of the queue to connect to (optional)
NSNumber* *workers = @1; // The number of workers to generate  (optional) (default to @1)
NSNumber* *useSSL = @56; // Use SSL (optional)

OAIAMQPApi *apiInstance = [[OAIAMQPApi alloc] init];

// Create Consumer
[apiInstance consumerCreateWithVersion:version
    appKey:appKey
    name:name
    hostname:hostname
    username:username
    password:password
    dataMapping:dataMapping
    deviceId:deviceId
    accountId:accountId
    port:port
    virtualHost:virtualHost
    exchanger:exchanger
    exchangerType:exchangerType
    workers:workers
    useSSL:useSSL
              completionHandler: ^(OAIQueueResponse* output, NSError* error) {
                            if (output) {

                            }
                        }];
```
